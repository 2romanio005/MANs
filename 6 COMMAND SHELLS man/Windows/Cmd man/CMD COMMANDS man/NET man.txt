help <директива>							- справка по директиве

use											- подключить общую папку в качестве диска
	/PERSISTENT:<NO | YES>					- запомнить (незапоминать) сетевые подключения
	local /DELETE							- удаляет подключение не сохраняя
	/USER
	/SAVECRED								- указывает, что имя пользователя и пароль следует сохранить. Этот параметр пропускается, если команда не запрашивает имя пользователя и пароль
	/HOME									- Подключает пользователя к домашнему каталогу
	
	net use R: "\\192.168.1.145\4 semester" /user:Администратор
	net use * "\\192.168.1.145\4 semester" /user:Roman Password

view										- отображает список общих ресурсов компьютера. При использовании без параметров отображается список всех компьютеров в текущем домене или рабочей группе

stop/start									- запус остановка службы

share										- управление использованием серверных ресурсов другим пользователям сети
	/GRANT:пользователь								- устанавливает разрешения для указанного пользователя
	/USERS:число									- устанавливает максимальное число пользователей
	/UNLIMITED										- снимает ограничение на число пользователей, одновременно использующих общий ресурс
	/CACHE											- настройка кеширования
	name /DELETE
	
	new share name=path /GRANT:user1,Full						- создать общую папку с именем name по пути path и предоставить оплный доступ user1 к нему

config <SERVER | WORKSTATION>				- отображает информацию о настройке служб станции или службы сервера.
 
session										- управление текущими подключениями
	/list
	ip(name) /delete
	
user										- управление пользователями	
	/add
	/del
	name *									- ввести следующей командой пароль
	name password							- установить такой пароль

statistics WORKSTATION						- выводит журнал статистики для локальной службы рабочей станции или службы сервера

localgroup имя_группы						- управление группами пользователей
	/ADD
	/DEL
