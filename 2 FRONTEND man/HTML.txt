1.7
<table>						- таблица
	<caption></caption>		- заголовок всей таблицы
	<colgroup>				- стилизация по колонкам
		<col />				- одна колонка (указываем стили)
		<col span="2" />	- обединение двух столбцов одним стилем (2 с 3 т.к. 1 задан в пред строчке)
	</colgroup>
	<tr>					- одна строка таблицы
		<th></th>			- заголовок строк или сболбцов (как td но жирный)
		<td></td>			- один элемент таблицы
			rowspan="2"		- атрибут для td и th чтобы объединить два элемента из одной колонки
			colspan="2"		- атрибут для td и th чтобы объединить два элемента из одной строки
	</tr>
</table>

<table>						- таблица со структурой
	<thead></thead>			- все заголовки сверху
	<tbody></tbody>			- тело таблицы
	<tfoot></tfoot>			- нижняя строка в которой какие то общие данные по каждому столбцу (например сумма)
</table>
